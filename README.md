      def Rmat(degree): #degree 매개변수로 받은 각도 -> radian 변환 (행렬 회전 연산을 위해)
          rad = np.deg2rad(degree) 
          c = np.cos(rad)
          s = np.sin(rad)
          R = np.array([ [c, -s, 0],
                         [s,  c, 0], [0,0,1]])
          return R
      
      def Tmat(tx, ty):
          Translation = np.array( [
              [1, 0, tx],
              [0, 1, ty],
              [0, 0, 1]
          ])
          return Translation
